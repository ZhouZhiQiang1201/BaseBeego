导包

    github.com/astaxie/beego/toolbox

    
task任务

    任务名称  spec定时任务格式  执行函数
    tk1 := toolbox.NewTask("tk1", "0 12 * * * *", func() error { fmt.Println("tk1"); return nil })

    
测试

    err := tk1.Run()
    if err != nil {
        t.Fatal(err)
    }

    
添加到计划任务表

    toolbox.AddTask("tk1", tk1)

    
开始执行全局任务

    toolbox.StartTask()
    defer toolbox.StopTask()
